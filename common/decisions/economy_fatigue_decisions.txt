			}
		}

		is_good = no
		days_mission_timeout = 9

		complete_effect = {
			custom_effect_tooltip = iron_shortage_avoided_tt
		}

		timeout_effect = {
			custom_effect_tooltip = iron_shortage_tt
			set_temp_variable = { tempIron = resource@iron }
			set_temp_variable = { tempCoal = resource@coal }
			while_loop_effect = {
				limit = {
					OR = {
						check_variable = { tempIron < 0 }
						check_variable = { tempCoal < 0 }
					}
				}
				add_to_temp_variable = {
					var = tempIron
					value = 25
				}
				add_to_temp_variable = {
					var = tempCoal
					value = 20
				}
				random_controlled_state = {
					limit = {
						steel_refinery > 0
					}
					disable_steel_mill = yes
				}
			}
		}
	}
